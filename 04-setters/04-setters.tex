class BookDTO {
  private int id;
  private String author;
  private String title;
  BookDTO(int i, String a, String t)
    { id = i; author = a; title = t; }
  int getId() { return id; }
  String getAuthor() { return author; }
  String getTitle() { return title; }
}
\end{ffcode}
}
